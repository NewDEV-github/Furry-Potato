305func align_name_label():
306 var name_padding = _theme.get_value('name', 'name_padding', Vector2( 10, 0 ))
307 var horizontal_offset = _theme.get_value('name', 'horizontal_offset', 0)
308 var name_label_position = _theme.get_value('name', 'position', 0)
309 var label_size = name_label.rect_size.x
310 if name_label_position == 0:
311 name_label.rect_global_position.x = rect_global_position.x + horizontal_offset
312 elif name_label_position == 1: # Center
