171func godot_dialog_connect(who, method_name, signal_name = "file_selected"):
172 # You can pass multiple signal_name using an array
173
174 # Checking if previous connections exist, if they do, disconnect them.
175 for test_signal in editor_file_dialog.get_signal_list():
176 if not file_picker_data['node'] or not is_instance_valid(file_picker_data['node']):
177 continue
178
179 if editor_file_dialog.is_connected(
180 test_signal.name,
181 file_picker_data['node'],
182 file_picker_data['method']
183 ):
184 editor_file_dialog.disconnect(
185 test_signal.name,
186 file_picker_data['node'],
187 file_picker_data['method']
188 )
189 # Connect new signals
190 for new_signal_name in signal_name if typeof(signal_name) == TYPE_ARRAY else [signal_name]:
191 editor_file_dialog.connect(new_signal_name, who, method_name, [who])
192
193 file_picker_data['method'] = method_name
194 file_picker_data['node'] = who
