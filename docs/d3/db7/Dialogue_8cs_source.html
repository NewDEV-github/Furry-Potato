141 private async void DisplayLineByLine(DialogueLine dialogueLine) {
142 var lines = dialogueLine.Lines;
143 var currentLineNumber = 0;
144 foreach (var (lineId, text) in lines) {
145 _dialogueText.Show();
146 var characterName = "";
147 if (dialogueLine.Characters.Count == 1) {
148 characterName = dialogueLine.Characters[0];
149 }
150 else {
151 characterName = lineId != "CHOICE" ? dialogueLine.Characters[lineId.ToInt()] : "CHOICE";
152 characterName = lineId != "END" ? dialogueLine.Characters[lineId.ToInt()] : "END";
153 }
154
155 currentLineNumber++;
156 if (characterName == "CHOICE" || text == "CHOICE") {
157 ShowChoicesAfterCurrentLine(dialogueLine, currentLineNumber);
158 }
159 else {
160 _dialogueText.Text = text;
161 _dialogueCharacterName.Text = characterName;
162 }
163
164 await ToSignal(this, "EnterPressed");
165 if (characterName == "END" || text == "END") EndDialogue();
166 }
167 }
