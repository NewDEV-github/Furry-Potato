104static func parse_branches(current_dialog, dialog_script: Dictionary) -> Dictionary:
105 current_dialog.questions = [] # Resetting the questions
106
107 # Return the same thing if it doesn't have events
108 if dialog_script.has('events') == false:

111 var parser_queue = [] # This saves the last question opened, and it gets removed once it was consumed by a endbranch event
112 var event_idx: int = 0 # The current id for jumping later on
113 var question_idx: int = 0 # identifying the questions to assign options to it
114 for event in dialog_script['events']:
115 if event['event_id'] == 'dialogic_011':
116 var opened_branch = parser_queue.back()
117 var option = {
118 'question_idx': opened_branch['question_idx'],

126 'event_idx': event_idx,
127 'condition': event['condition'],
128 'definition': event['definition'],
129 'value': event['value'],
130 }
131 else:
132 option = {
133 'question_idx': opened_branch['question_idx'],
134 'label': parse_definitions(current_dialog, event['choice'], true, false),
135 'event_idx': event_idx,
136 'condition': '',
137 'definition': '',
138 'value': '',

159 event['end_branch_of'] = opened_branch['question_idx']
160 dialog_script['events'][opened_branch['event_idx']]['end_idx'] = event_idx
