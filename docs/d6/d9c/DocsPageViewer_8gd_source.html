140func _on_meta_clicked(meta):
141 ## Check wether this is a real LINK
142 if meta.begins_with("http"):
143
144 # test if we can interpret this as a normal link to a docs file
145 if meta.count("Documentation/Content") > 0:
146 meta = meta.split("Documentation/Content")[1]
147
148 # else open it with the browser
149 else:
150 OS.shell_open(meta)
151 return
152
153 ## Check wether it is a section
154 if meta.begins_with("#"):
155 # try to open it in this document
156 scroll_to_section(meta)
157
158 ## Else send a signal that the pluginmaker has to interpret
159 else:
160 # if the link contains a section
161 var link = meta
162 var section = null
163 if meta.count("#") > 0:
164 var split = meta.split('#')
165 link = split[0]
166 section = split[1]
167 if link.begins_with('.'):
168 link = current_path.trim_suffix(current_path.get_file()).trim_suffix("/") + link.trim_prefix(".")
169 if not link.begins_with("res://"):
170 link = documentation_path.plus_file('Content').plus_file(link)
171 if not link.ends_with(".md"):
172 link += '.md'
173
