38func play(from_position=0.0, playing_sample_nb=-1):
39 var number_of_samples = len(samples)
40 if number_of_samples > 0:
41 if playing_sample_nb < 0:
42 if number_of_samples == 1:
43 playing_sample_nb = 0
44 else:
45 match random_strategy:
46 1:
47 playing_sample_nb = randi() % (number_of_samples - 1)
48 if last_played_sample_nb == playing_sample_nb:
49 playing_sample_nb += 1
50 last_played_sample_nb = playing_sample_nb
51 2:
52 if len(to_play) == 0:
53 for i in range(number_of_samples):
54 if i != last_played_sample_nb:
55 to_play.append(i)
56 to_play.shuffle()
57 playing_sample_nb = to_play.pop_back()
58 last_played_sample_nb = playing_sample_nb
59 _:
60 playing_sample_nb = randi() % number_of_samples
61 if random_volume_range != 0:
62 .set_volume_db(base_volume + (randf() - .5) * random_volume_range)
63 if random_pitch_range != 0:
64 .set_pitch_scale(max(0.0001, base_pitch + (randf() - .5) * random_pitch_range))
65 set_stream(samples[playing_sample_nb])
66 .play(from_position)
