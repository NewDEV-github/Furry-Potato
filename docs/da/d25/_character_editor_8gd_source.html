231# is called recursively to build all levels of the folder structure
232func build_PickerMenuFolder(menu:PopupMenu, folder_structure:Dictionary, current_folder_name:String):
233 var index = 0
234
235 if menu == nodes["theme"].get_popup():
236 menu.add_item('No custom theme')
237 menu.set_item_icon(index, get_icon("GuiRadioUnchecked", "EditorIcons"))
238 menu.set_item_metadata(index, {"file":""})
239 index += 1
240
241 for folder_name in folder_structure['folders'].keys():
242 var submenu = PopupMenu.new()
243 var submenu_name = build_PickerMenuFolder(submenu, folder_structure['folders'][folder_name], folder_name)
244 submenu.name = submenu_name
245 menu.add_submenu_item(folder_name, submenu_name)
246 menu.set_item_icon(index, get_icon("Folder", "EditorIcons"))
247 menu.add_child(submenu)
248 index += 1
249
250 # give it the right style
251 nodes['theme'].update_submenu_style(submenu)
252
253 var files_info = DialogicUtil.get_theme_dict()
254 for file in folder_structure['files']:
255 menu.add_item(files_info[file]['name'])
256 menu.set_item_icon(index, editor_reference.get_node("MainPanel/MasterTreeContainer/MasterTree").theme_icon)
257 menu.set_item_metadata(index, {'file':file})
258 index += 1
259
260 if not menu.is_connected("index_pressed", self, "_on_theme_selected"):
261 menu.connect("index_pressed", self, '_on_theme_selected', [menu])
262
263 return current_folder_name

338func _on_dir_selected(path, target):
339 var dir = Directory.new()
340 if dir.open(path) == OK:
341 dir.list_dir_begin()
342 var file_name = dir.get_next()
343 while file_name != "":
344 if not dir.current_is_dir():
345 var file_lower = file_name.to_lower()
346 if '.svg' in file_lower or '.png' in file_lower:
347 if not '.import' in file_lower:
348 var final_name = path+ "/" + file_name
349 create_portrait_entry(DialogicResources.get_filename_from_path(file_name), final_name)
350 file_name = dir.get_next()
351 else:
352 print("An error occurred when trying to access the path.")
