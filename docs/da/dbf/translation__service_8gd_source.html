19static func get_translations() -> Dictionary:
20 var translations_resources = ['en', 'zh_CN', 'es', 'fr', 'de']
21 var translations = {}
22
23 for resource in translations_resources:
24 var t:PHashTranslation = load('res://addons/dialogic/Localization/dialogic.' + resource + '.translation')
25 if translations.has(t.locale):
26 translations[t.locale].append(t)
27 else:
28 translations[t.locale] = [t]

32static func _get_translation(message)->String:
33 var returned_translation = message
34 var translations = get_translations()
35 var default_fallback = 'en'
36
37 var editor_plugin = EditorPlugin.new()
38 var editor_settings = editor_plugin.get_editor_interface().get_editor_settings()
39 var locale = editor_settings.get('interface/editor/editor_language')
40
41 var cases = translations.get(
42 locale,
43 translations.get(default_fallback, [PHashTranslation.new()])
44 )
45 for case in cases:
46 returned_translation = (case as PHashTranslation).get_message(message)
47 if returned_translation:
48 break
49 else:
50 # If there's no translation, returns the original string
51 returned_translation = message
52
53 #print('Message: ', message, ' - locale: ', locale, ' - ', returned_translation)
54 return returned_translation
