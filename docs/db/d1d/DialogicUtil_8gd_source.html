133static func compare_definitions(def_value: String, event_value: String, condition: String):
134 var definitions
135 if not Engine.is_editor_hint():
136 if Engine.get_main_loop().has_meta('definitions'):
137 definitions = Engine.get_main_loop().get_meta('definitions')
138 else:
139 definitions = DialogicResources.get_default_definitions()
140 Engine.get_main_loop().set_meta('definitions', definitions)
141 else:
142 definitions = DialogicResources.get_default_definitions()
143 var condition_met = false
144 if def_value != null and event_value != null:
145 # check if event_value equals a definition name and use that instead
146 for d in definitions['variables']:
147 if (d['name'] != '' and d['name'] == event_value):
148 event_value = d['value']
149 break;
150 var converted_def_value = def_value
151 var converted_event_value = event_value
152 if def_value.is_valid_float() and event_value.is_valid_float():
153 converted_def_value = float(def_value)
154 converted_event_value = float(event_value)
155 if condition == '':
156 condition = '==' # The default condition is Equal to
157 match condition:
158 "==":
159 condition_met = converted_def_value == converted_event_value
160 "!=":
161 condition_met = converted_def_value != converted_event_value
162 ">":
163 condition_met = converted_def_value > converted_event_value
164 ">=":
165 condition_met = converted_def_value >= converted_event_value
166 "<":
167 condition_met = converted_def_value < converted_event_value
168 "<=":
169 condition_met = converted_def_value <= converted_event_value
170 return condition_met

359static func check_folders_recursive(folder_data: Dictionary, file_names:Array):
360 if not folder_data.has('metadata'):
361 folder_data['metadata'] = {'color':null, 'folded':false}
362 for folder in folder_data['folders'].keys():
363 var result = check_folders_recursive(folder_data["folders"][folder], file_names)
364 folder_data['folders'][folder] = result[0]
365 file_names = result[1]
366 for file in folder_data['files']:
367 if not file in file_names:
368 folder_data["files"].erase(file)
369 #print("[D] The file ", file, " was deleted!")
370 else:
371 file_names.erase(file)
372 return [folder_data, file_names]

606 static func key_available(key, a: Dictionary) -> bool:

609 static func get_compare_value(a: Dictionary) -> String:
610 if key_available('display_name', a):
611 return a['display_name']
612
613 if key_available('name', a):
614 return a['name']
615
616 if key_available('id', a):
617 return a['id']
618
619 if 'metadata' in a.keys():
620 var a_metadata = a['metadata']
621 if key_available('name', a_metadata):
622 return a_metadata['name']
623 if key_available('file', a_metadata):
