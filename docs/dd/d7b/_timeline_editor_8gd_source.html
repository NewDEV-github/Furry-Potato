529func select_item(item: Node, multi_possible:bool = true):
530 if item == null:
531 return
532
533 if Input.is_key_pressed(KEY_CONTROL) and multi_possible:
534 # deselect the item if it is selected
535 if _is_item_selected(item):
536 selected_items.erase(item)
537 else:
538 selected_items.append(item)
539 elif Input.is_key_pressed(KEY_SHIFT) and multi_possible:
540
541 if len(selected_items) == 0:
542 selected_items = [item]
543 else:
544 var index = selected_items[-1].get_index()
545 var goal_idx = item.get_index()
546 while true:
547 if index < goal_idx: index += 1
548 else: index -= 1
549 if not timeline.get_child(index) in selected_items:
550 selected_items.append(timeline.get_child(index))
551
552 if index == goal_idx:
553 break
554 else:
555 if len(selected_items) == 1:
556 if _is_item_selected(item):
557 selected_items.erase(item)
558 else:
559 selected_items = [item]
560 else:
561 selected_items = [item]
