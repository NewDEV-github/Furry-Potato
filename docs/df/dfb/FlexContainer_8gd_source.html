4# The flow container will fit as many children in a row as it can
5# using their minimum size, and will then continue on the next row.
6# Does not use SIZE_EXPAND flags of children.

30func _get_minimum_size() -> Vector2:
31 var max_child_width: float = 0
32
33 for child in get_children():
34 if not child.has_method("get_combined_minimum_size"):
35 break
36
37 var requested_size: Vector2 = child.get_combined_minimum_size()
38 if requested_size.x > max_child_width:
39 max_child_width = requested_size.x
40
41 var height := _calculate_layout(false)
42 _reported_height_at_last_minimum_size_call = height
43
44 return Vector2(max_child_width, height)

47func _notification(what):
48 if (what==NOTIFICATION_SORT_CHILDREN):
49 var height = _calculate_layout(true)
50
51 if height != _reported_height_at_last_minimum_size_call:
52 _make_parent_reevaluate_our_size()

54# If apply is true, the children will actually be moved to the calculated
55# locations.
56# Returns the resulting height.
57func _calculate_layout(apply: bool) -> float:
58 var child_position: Vector2 = Vector2(0, 0)
59 var row_height: float = 0
60 var container_width: float = rect_size.x
61 var num_children_in_current_row: float = 0
62
63 for child in get_children():
64 if not child.has_method("get_combined_minimum_size"):
65 continue
66 if not child.visible:
67 continue
68
69 var child_min_size: Vector2 = child.get_combined_minimum_size()
70
71 if num_children_in_current_row > 0:
72 child_position.x += horizontal_margin
73
74 if child_position.x + child_min_size.x > container_width:
75 # Go to the next row.
76 child_position = Vector2(0, child_position.y + row_height + vertical_margin)
77 row_height = 0
78 num_children_in_current_row = 0
79
80 if apply:
81 fit_child_in_rect(child, Rect2(child_position, child_min_size))
82
83 if child_min_size.y > row_height:
84 row_height = child_min_size.y
85
86 child_position.x += child_min_size.x
87 num_children_in_current_row += 1
88
89 return child_position.y + row_height
